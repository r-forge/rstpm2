## The function is currently defined as
function (var, increment = 1) 
{
    n <- length(var)
    if (n > 1) 
        increment <- rep(increment, n)
    function(data) {
        for (i in 1:n) {
            data[[var[i]]] <- data[[var[i]]] + increment[i]
        }
        data
    }
  }
